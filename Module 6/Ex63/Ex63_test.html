function validateForm() { var name = document.getElementById("name").value; var email = document.getElementById("email").value; var gender = document.querySelector('input[name="gender"]:checked'); var hobbies = document.querySelectorAll('input[name="hobbies"]:checked'); var color = document.querySelector('input[name="color"]:checked'); if (name === "") { alert("Name cannot be left blank"); return false; } if (email === "") { alert("Email cannot be left blank"); return false; } if (gender === null) { alert("Please select a gender"); return false; } if (hobbies.length === 0) { alert("Please select at least one hobby"); return false; } if (color === null) { alert("Please select a favorite color"); return false; } displayFormData(name, email, gender.value, hobbies, color.value); } function displayFormData(name, email, gender, hobbies, color) { var table = document.getElementById("tbl"); var newRow = table.insertRow(-1); var nameCell = newRow.insertCell(0); var emailCell = newRow.insertCell(1); var genderCell = newRow.insertCell(2); var hobbiesCell = newRow.insertCell(3); var colorCell = newRow.insertCell(4); nameCell.innerText = name; emailCell.innerText = email; genderCell.innerText = gender; hobbiesCell.innerText = Array.from(hobbies).map(hobby => hobby.value).join(', '); colorCell.innerText = color; resetForm(); } function resetForm() { document.getElementById("name").value = ""; document.getElementById("email").value = ""; document.querySelector('input[name="gender"]:checked').checked = false; document.querySelectorAll('input[name="hobbies"]:checked').forEach(hobby => hobby.checked = false); document.querySelector('input[name="color"]:checked').checked = false; document.getElementById("name").focus(); }
